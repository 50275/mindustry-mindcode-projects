// Name: Block Core Filler
// Fills all connected blocks to 50% item capacity of all resources.  

for i in 0..@links do 
    firstcore = getlink(i); 
    
    capacity = firstcore.@itemCapacity; 
    f = capacity \ 2; 
    
    // set sharded items to half of every resource
    for j in 0...@itemCount do
        material = lookup(:item, j); 
        firstcore.setprop(material, f); 
    end; 
end;
