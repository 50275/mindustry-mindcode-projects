def getUnit(type, flag)
    ubind(type);
    if @unit == null then
        return 0; 
    else
        first = @unit; 
        if first != null && first.@flag == flag then
            return @unit; 
        else 
            do 
                ubind(type);
            loop while (@unit == null || @unit.@flag != flag) && @unit != first; 
            if @unit == first then
                return 0; 
            end;
        end; 
    end; 
    return @unit; 
end; 
def countUnits (type)
    count = 0;
    ubind(type);
    if @unit != null then
        first = @unit;
        while !first.@dead do 
            count+=1; 
            ubind(type);
            if @unit == first then
                break;
            end;
        end;
    end;
    return count; 
end;
